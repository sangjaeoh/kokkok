CREATE TABLE BOARD
(
    SEQ            NUMBER           NOT NULL, 
    BCODE          NUMBER           NULL, 
    USERID             VARCHAR2(20)     NULL, 
    SUBJECT        VARCHAR2(100)    NULL, 
    CONTENT        CLOB             NULL, 
    LOGTIME        DATE             NULL, 
    UPDATETIME     DATE             NULL, 
    RECOMMCOUNT    NUMBER           NULL, 
    WISHCOUNT      NUMBER           NULL, 
    HIT            NUMBER           NULL, 
    CONSTRAINT BOARD_PK PRIMARY KEY (SEQ)
)
/

CREATE SEQUENCE BOARD_SEQ
START WITH 1
INCREMENT BY 1;
/
